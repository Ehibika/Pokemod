actor ClenseSeed : ConsumableBase 
{
inventory.amount 1
inventory.maxamount 5
inventory.pickupmessage "$PU_CLENSESEED"
Tag "$TAG_CLENSESEED"
inventory.icon "CLSEEDI"
Inventory.PickupSound "Misc/GetItem"
+INVBAR
+COUNTITEM
Translation "250:250=214:214", "192:192=229:229", "198:198=230:230"
scale 2.0
States
{
Spawn:
PITM B 0 
PITM B 6
Goto Spawn
use1:
TNT1 A 0 A_Jumpifinventory("SleepMark",1,"Success")
TNT1 A 0 A_Jumpifinventory("ParalysisMark",1,"Success")
TNT1 A 0 A_Jumpifinventory("BurnMark",1,"Success")
TNT1 A 0 A_Jumpifinventory("PoisonMark",1,"Success")
Goto Success2

Success:
TNT1 A 0 A_TakeInventory("SleepMark",99)
TNT1 A 0 A_TakeInventory("ParalysisMark",99)
TNT1 A 0 A_TakeInventory("BurnMark",99)
TNT1 A 0 A_TakeInventory("PoisonMark",99)
TNT1 A 0 A_GiveInventory("CureAll",1)
TNT1 A 0 A_PlaySound("Support/Heal1")
Goto Success2
Success2:
TNT1 A 0 A_SpawnitemEX("ConsumeFX",0,0,0,0,0,0,0,1)
stop
}
}
