actor MagicalLeafMove : WepMasterBase
{
Tag "MagicalLeaf"
obituary "$OB_MagicalLeaf"
Weapon.AmmoUse 1
Weapon.AmmoGive 20
weapon.ammotype "MagicalLeafPP"
scale 2.0
Inventory.Pickupmessage "$PU_MEGABUSTER"
States
{
Spawn:
WEAP X 1
stop


Fire:
"####" B 0 //A_Jumpif(CallACS("CheckWeaponFire",RC_A)==0,"NoAmmo")
"####" B 0 A_Jump(256,"Fire.Bullet")
Goto Fire.Bullet

Fire.Bullet:
"####" B 0 A_Jumpif(CallACS("CheckWeaponFire",RC_A)==0,"NoAmmo")
"####" B 0 A_PlaySoundEx("weapon/tengushot","Weapon")
"####" B 0 A_FireCustomMissile("MagicalLeafShot",0,1,0,0)
"####" B 0 A_Jump(256,"Fire.After")
Goto Fire.After

Fire.After:
"####" B 1 A_GiveInventory("HalfRecovery_A")
"####" C 0 A_REfire("Fire")
"####" B 0 A_Jump(256,"Fire.End")
Goto Fire.End





}
}

actor MagicalLeafShot : BasicProjectile
{
Speed 35
Damage (15)
DamageType "MagicalLeaf"
obituary "$OB_MagicalLeaf"
+SEEKERMISSILE
+SCREENSEEKER
radius 5
Reactiontime 15
height 16
scale 3.5
States
{
Spawn:
TNT1 A 0 
TNT1 A 0 A_GiveInventory("ProjectileCode",1)
TNT1 A 0 A_SpawnItemEx("MagicalLeafFX",0,0,0,0,0,0,0,1)
Spawn2: // -=- Seeking Target -=-
MLEF B 1 A_SeekerMissile (2, 10, SMF_LOOK|SMF_PRECISE, 256, 2)
MLEF B 1 A_SpawnItemEx("AquaRingSparkle",random(-10,10),random(-10,10),random(-10,10),0,0,0,0,1)
MLEF A 0 A_Countdown
MLEF C 1 A_SpawnItemEx("AquaRingSparkle",random(-10,10),random(-10,10),random(-10,10),0,0,0,0,1)
MLEF C 1 A_SeekerMissile (2, 10, SMF_LOOK|SMF_PRECISE, 256, 2)
MLEF A 0 A_Countdown
MLEF D 1 A_SpawnItemEx("AquaRingSparkle",random(-10,10),random(-10,10),random(-10,10),0,0,0,0,1)
MLEF D 1 A_SeekerMissile (2, 10, SMF_LOOK|SMF_PRECISE, 256, 2)
MLEF A 0 A_Countdown

TNT1 A 0 A_JumpIf(IsPointerEqual(AAPTR_TRACER, AAPTR_NULL) == TRUE, "Spawn2")
Goto Spawn3

Spawn3: // -=- ExamineTarget -=-
TNT1 A 0 A_JumpifInventory("SafeguardPower",1,"TargClear",AAPTR_TRACER)
Goto Spawn4

TargClear:
MLEF A 0 A_RearrangePointers(AAPTR_DEFAULT, AAPTR_DEFAULT, AAPTR_NULL)
Goto Spawn4

Spawn4: // -=- target acquired -=-
TNT1 A 0 A_SeekerMissile (2, 10, SMF_PRECISE)
MLEF B 2 A_SpawnItemEx("AquaRingSparkle",random(-10,10),random(-10,10),random(-10,10),0,0,0,0,1)
TNT1 A 0 A_SeekerMissile (2, 10, SMF_PRECISE)
MLEF C 2 A_SpawnItemEx("AquaRingSparkle",random(-10,10),random(-10,10),random(-10,10),0,0,0,0,1)
TNT1 A 0 A_SeekerMissile (2, 10, SMF_PRECISE)
MLEF D 2 A_SpawnItemEx("AquaRingSparkle",random(-10,10),random(-10,10),random(-10,10),0,0,0,0,1)
MLEF A 0 A_Countdown
loop


Death:
TNT1 A 0 A_GiveInventory("ProjectileDCode",1)
BSTG AAAAAA 0 A_SpawnItemEx("MagicalLeafParticle", 0, 0, 28, 256/24, 256/24, random2(256/24), random(0, 359),1)
BSTG AAAAAA 0 A_SpawnItemEx("AquaRingSparkle", 0, 0, 28, 256/24, 256/24, random2(256/24), random(0, 359),1)
SPKH ABCD 2
stop
}
}
